    <script>
        const data = {
            genesis: {
                title: "GENESIS LINEUP",
                cars: ["G70", "G80", "G90", "GV70", "GV80"],
                link: "https://www.genesis.com/kr/ko/view-delivery/estimate/models.html"
            },
            hyundai: {
                title: "HYUNDAI LINEUP",
                cars: ["아반떼", "쏘나타", "그랜저", "베뉴", "코나", "투싼", "싼타페", "팰리세이드", "포터", "스타리아"],
                link: "https://www.hyundai.com/kr/ko/e/vehicles/build-your-own"
            }
        };

        let currentBrand = '';

        function showBrand(brand) {
            currentBrand = brand;
            document.getElementById('main-screen').classList.add('hidden');
            document.getElementById('list-screen').classList.remove('hidden');
            document.getElementById('brand-title').innerText = data[brand].title;
            
            const listDiv = document.getElementById('car-list');
            listDiv.innerHTML = '';
            
            data[brand].cars.forEach(car => {
                const btn = document.createElement('div');
                btn.className = "bg-white p-6 text-center shadow rounded-lg font-bold border hover:bg-gray-50";
                btn.innerText = car;
                // 상세 페이지 연결 (현대차 공식 페이지의 검색 결과로 연결되게 설정)
                btn.onclick = () => window.open(`https://www.hyundai.com/kr/ko/e/search?searchKeyword=${car}`);
                listDiv.appendChild(btn);
            });
        }

        function goBack() {
            document.getElementById('main-screen').classList.remove('hidden');
            document.getElementById('list-screen').classList.add('hidden');
        }

        function openEstimate() {
            window.open(data[currentBrand].link);
        }

        function shareContact() {
            // 실제 영업사원님의 정보를 아래에 기입하세요
            const name = "홍길동 카마스터";
            const phone = "010-1234-5678";
            const text = `안녕하세요! 현대자동차 ${name}입니다. 상담이 필요하시면 언제든 연락주세요.\n전화: ${phone}`;
            
            if (navigator.share) {
                navigator.share({ title: '명함 공유', text: text });
            } else {
                alert("복사되었습니다: " + text);
            }
        }
    </script>
